184  inline void update( int i, axis_t partialGrad ){
185 
186  /* Gradient update */
187  grad[0] += partialGrad.x;
188  grad[1] += partialGrad.y;
189  grad[2] += partialGrad.z;
190 
191  /* Gradient IGM update */
192  gradIGM[0] -= std::abs(partialGrad.x);
193  gradIGM[1] -= std::abs(partialGrad.y);
194  gradIGM[2] -= std::abs(partialGrad.z);
195 
196  /* Atoms belong to the first molecule */
197  if(i < nbAtomInMolA){
198 
199  /* Contribution inside molecule A update */
200  contributionInsideMoleculeA[0] += partialGrad.x;
201  contributionInsideMoleculeA[1] += partialGrad.y;
202  contributionInsideMoleculeA[2] += partialGrad.z;
203 
204  /* Atoms belong to the second molcule */
205  }else{
206 
207  /* Contribution inside molecule B update */
208  contributionInsideMoleculeB[0] += partialGrad.x;
209  contributionInsideMoleculeB[1] += partialGrad.y;
210  contributionInsideMoleculeB[2] += partialGrad.z;
211  }
212 
213  /* Updating values for the atoms */
214  contributionAtomicAlone[i].x = partialGrad.x;
215  contributionAtomicAlone[i].y = partialGrad.y;
216  contributionAtomicAlone[i].z = partialGrad.z;
217 
218 
219  }
