240  inline void validate(PREC * increments, PREC ligandRadius = -1){
241 
242 
243  /* convert increment from angstroms to Bohr */
244  increments[0] = increments[0] / BOHRTOA;
245  increments[1] = increments[1] / BOHRTOA;
246  increments[2] = increments[2] / BOHRTOA;
247 
248 
249  /* If CUBE and RADIUS have not been used */
250  if( !boxSet && !radiusSet){
251 
252  /* If a ligand has been provided */
253  if( numLigand != -1 ){
254 
255 
256  /* Updating max coordinates */
257  maxCoordinates[0] += ligandRadius / BOHRTOA;
258  maxCoordinates[1] += ligandRadius / BOHRTOA;
259  maxCoordinates[2] += ligandRadius / BOHRTOA;
260 
261  /* Updating min coordinates */
262  minCoordinates[0] -= ligandRadius / BOHRTOA;
263  minCoordinates[1] -= ligandRadius / BOHRTOA;
264  minCoordinates[2] -= ligandRadius / BOHRTOA;
265 
266  /* If no ligand has been provided */
267  }else{
268 
269 
270 
271  /* Updating max coordinates */
272  maxCoordinates[0] += 2.f;
273  maxCoordinates[1] += 2.f;
274  maxCoordinates[2] += 2.f;
275 
276  /* Updating min coordinates */
277  minCoordinates[0] -= 2.f;
278  minCoordinates[1] -= 2.f;
279  minCoordinates[2] -= 2.f;
280 
281  } // end of else of if( numLigand != -1 ){
282 
283  } // end of if( !boxSet && !radiusSet){
284 
285  /* Setting the steps */
286  nbSteps[0] = static_cast<int>( std::ceil( (maxCoordinates[0] - minCoordinates[0]) / increments[0] ) );
287  nbSteps[1] = static_cast<int>( std::ceil( (maxCoordinates[1] - minCoordinates[1]) / increments[1] ) );
288  nbSteps[2] = static_cast<int>( std::ceil( (maxCoordinates[2] - minCoordinates[2]) / increments[2] ) );
289 
290  if(cursor != nbAtom){
291  std::cerr << "The number of atom read is different from the number of atom that should have been read. Please verify the .xyz files." << std::endl;
292  std::cerr << "The program will now exit." << std::endl;
293  exit(EXIT_FAILURE);
294 
295  }
