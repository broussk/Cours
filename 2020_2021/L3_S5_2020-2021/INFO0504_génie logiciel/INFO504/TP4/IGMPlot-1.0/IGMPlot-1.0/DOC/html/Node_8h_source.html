184  void computeDensity(position_t & __restrict posGrid, param_t & __restrict params){
185 
186 
187  /* Initilization */
188  rho = 0.f;
189  contribution = 0.f;
190 
191 
192  /* Looping through the atoms in the first molecule */
193  for( int j(0) ; j < nbAtomMolA ; ++j ){
194 
195  int atomType = pdata.atomTypes[j];
196  PREC pos[3] = {pdata.atomPositions.xValues[j], pdata.atomPositions.yValues[j], pdata.atomPositions.zValues[j]};
197 
198  /* Setting differences */
199  diffX[j] = ( posGrid.x - pos[X] );
200  diffY[j] = ( posGrid.y - pos[Y] );
201  diffZ[j] = ( posGrid.z - pos[Z] );
202 
203  /* Processing differences' squares */
204  squareX[j] = ( diffX[j] * diffX[j] );
205  squareY[j] = ( diffY[j] * diffY[j] );
206  squareZ[j] = ( diffZ[j] * diffZ[j] );
207 
208  /* Distance processing */
209  distance[j] = std::sqrt( squareX[j] + squareY[j] + squareZ[j] );
210 
211  /* Setting R_TRESHOLD as minimum */
212  distance[j] = ( distance[j] < R_TRESHOLD ? R_TRESHOLD : distance[j] );
213 
214  /* Rho processing */
215  rhoX[j] = A1[ atomType ] * EXP( -B1[ atomType ] * distance[j] );
216  rhoY[j] = A2[ atomType ] * EXP( -B2[ atomType ] * distance[j] );
217  rhoZ[j] = A3[ atomType ] * EXP( -B3[ atomType ] * distance[j] );
218  rho += rhoX[j] + rhoY[j] + rhoZ[j];
219 
220  }
221 
222  /* Setting contribution as the rho for the first molecule */
223  contribution = rho;
224 
225 
226  /* Looping through the atoms in the first molecule */
227  for( int j(nbAtomMolA) ; j < nbAtom ; ++j){
228 
229  int atomType = pdata.atomTypes[j];
230  PREC pos[3] = {pdata.atomPositions.xValues[j], pdata.atomPositions.yValues[j], pdata.atomPositions.zValues[j]};
231 
232  /* Setting differences */
233  diffX[j] = ( posGrid.x - pos[X] );
234  diffY[j] = ( posGrid.y - pos[Y] );
235  diffZ[j] = ( posGrid.z - pos[Z] );
236 
237  /* Processing differenes' squares */
238  squareX[j] = ( diffX[j] * diffX[j] );
239  squareY[j] = ( diffY[j] * diffY[j] );
240  squareZ[j] = ( diffZ[j] * diffZ[j] );
241 
242  /* Distance processing */
243  distance[j] = std::sqrt( squareX[j] + squareY[j] + squareZ[j] );
244 
245  /* Setting R_TRESHOLD as minimum */
246  distance[j] = ( distance[j] < R_TRESHOLD ? R_TRESHOLD : distance[j] );
247 
248  /* Rho processing */
249  rhoX[j] = A1[ atomType ] * EXP( -B1[ atomType ] * distance[j] );
250  rhoY[j] = A2[ atomType ] * EXP( -B2[ atomType ] * distance[j] );
251  rhoZ[j] = A3[ atomType ] * EXP( -B3[ atomType ] * distance[j] );
252  rho += rhoX[j] + rhoY[j] + rhoZ[j];
253 
254  }
255 
256  /* Deciding whether the computing should continue or not depending on the contribution and rho */
257  contribution /= rho;
258  shouldCompute = (contribution <= params.intermolecular ) && (contribution >= (1-params.intermolecular));
259

266  void computeGradHess(){
267 
268 
269  /*Initializing */
270  data->init();
271  hess[0]=0.f;
272  hess[1]=0.f;
273  hess[2]=0.f;
274  hess[3]=0.f;
275  hess[4]=0.f;
276  hess[5]=0.f;
277 
278  for(int i = 0 ; i < nbAtom ; ++i){
279 
280  /* Computing the multiplicative inverse and its square */
281  multInvR = CAST_PREC(1.0 / static_cast<double>(distance[i]));
282  squareMultInvR = multInvR * multInvR;
283 
284 
285  /* Summing rho * B */
286  sumBRho = B1[pdata.atomTypes[i]] * rhoX[i] + B2[pdata.atomTypes[i]] * rhoY[i] + B3[pdata.atomTypes[i]] * rhoZ[i];
287 
288  /* Summing rho * B * B */
289  sumSquareBRho = B1[pdata.atomTypes[i]] * B1[pdata.atomTypes[i]] * rhoX[i]
290  + B2[pdata.atomTypes[i]] * B2[pdata.atomTypes[i]] * rhoY[i]
291  + B3[pdata.atomTypes[i]] * B3[pdata.atomTypes[i]] * rhoZ[i];
292 
293  /* Operation used three times */
294  opti = multInvR * sumBRho;
295 
296  /* Processing partial gradients */
297  partialGradient.x = -diffX[i] * opti;
298  partialGradient.y = -diffY[i] * opti;
299  partialGradient.z = -diffZ[i] * opti;
300 
301  /* Updating gradient values with current partial gradient values */
302  data->update(i, partialGradient);
303 
304  /* Updating hessian values */
305  hess[0] += squareMultInvR * ( (squareX[i] * multInvR - distance[i]) * sumBRho + squareX[i] * sumSquareBRho);
306  hess[3] += squareMultInvR * ( (squareY[i] * multInvR - distance[i]) * sumBRho + squareY[i] * sumSquareBRho);
307  hess[5] += squareMultInvR * ( (squareZ[i] * multInvR - distance[i]) * sumBRho + squareZ[i] * sumSquareBRho);
308 
309  /* Operation used three times */
310  opti += sumSquareBRho;
311  opti *= squareMultInvR;
312 
313  hess[1] += diffX[i] * diffY[i] * opti;
314  hess[2] += diffX[i] * diffZ[i] * opti;
315  hess[4] += diffY[i] * diffZ[i] * opti;
316  }

355  void computeSortLambdas(){
356 
357  a = -(hess[0] + hess[3] +hess[5]);
358 
359  b = hess[0] * hess[3] + hess[0] * hess[5] + hess[3] * hess[5] - hess[1] * hess[1] - hess[2] * hess[2] - hess[4] * hess[4];
360 
361  c = hess[0] * hess[4] * hess[4] + hess[3] * hess[2] * hess[2] + hess[5] * hess[1] * hess[1] - hess[0] * hess[3] * hess[5] - 2 * hess[1] * hess[2] * hess[4];
362 
363  /* Keeping 1/3 * a as the value is used five times */
364  opti = a * ONE_THIRD;
365 
366  p = b - a * opti;
367  q = a * a * a * 0.0740740740f - b * opti + c;
368 
369  determinant = q * q + p * p * p * 0.148148148148148148f;
370 
371  if(determinant < 0.f){
372 
373  /* These values are used three times */
374  opti2 = std::sqrt( -p * ONE_THIRD ) * 2.f;
375  opti3 = std::acos(3*q/(2*p) * std::sqrt(-3 / p));
376 
377  lambdas[0] = CAST_PREC( opti2 * cos( opti3 * ONE_THIRD) - opti);
378  lambdas[1] = CAST_PREC( opti2 * cos((opti3 + PI_2) * ONE_THIRD) - opti);
379  lambdas[2] = CAST_PREC( opti2 * cos((opti3 + PI_4) * ONE_THIRD) - opti);
380 
381  /* Sorting lambdas */
382  sortLambdas();
